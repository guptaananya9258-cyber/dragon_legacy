C++ CONCEPT
Inheritance

FireDragon, IceDragon, ForestDragon all inherit from abstract Dragon. They reuse its hp, attackPower, ability — IS-A relationship.

class FireDragon : public Dragon { public: FireDragon(string n) : Dragon(n,300,80, Ability("Inferno Blast",120,2)){} void attack() override; };
🔀
C++ CONCEPT
Polymorphism

Base class pointer Dragon* holds different subclass objects. At runtime, C++ calls the correct overridden attack() via vtable dispatch.

// Base pointer = POLYMORPHISM Dragon* player; if(choice==1) player=new FireDragon("Ember"); else player=new IceDragon("Frost"); player->attack(); // correct override! delete player;
🔒
C++ CONCEPT
Encapsulation

All data in Ability and Rider is private. Public methods provide the only controlled access — internal state is fully hidden.

class Ability { private: // HIDDEN string name; int damage, cooldown; int currentCooldown; public: // CONTROLLED bool canUse(); int getDamage(); };
📐
C++ CONCEPT
Abstraction

Dragon has pure virtual attack()=0. You cannot create a Dragon directly — only concrete subclasses. Hides complexity, forces the interface.

class Dragon { public: // PURE VIRTUAL (=0) = Abstraction virtual void attack() = 0; // Dragon d; ERROR — abstract! // FireDragon d; OK — concrete }; // COMPOSITION: Dragon HAS-A Ability Ability ability;

// ============================
// COMPOSITION: Ability class
// ENCAPSULATION: all private
// ============================
class Ability {
private: // ENCAPSULATION
    string name;
    int damage;
    int cooldown;
    int currentCooldown;
public:
    Ability(string n, int d, int cd)
        : name(n), damage(d), cooldown(cd), currentCooldown(0) {}

    bool canUse() { return currentCooldown == 0; }

    void use() {
        cout << "[Ability] " << name << " used! Damage: " << damage << endl;
        currentCooldown = cooldown;
    }

    void reduceCooldown() {
        if (currentCooldown > 0) currentCooldown--;
    }

    int getDamage() { return damage; }
};
// ============================================
// ABSTRACTION: abstract base class
// ENCAPSULATION: protected members
// COMPOSITION: Dragon HAS-A Ability
// ============================================
class Dragon {
protected: // accessible by child classes
    string name;
    int health, maxHealth;
    int attackPower;
    Ability ability; // COMPOSITION: HAS-A
public:
    Dragon(string n, int hp, int atk, Ability ab)
        : name(n), health(hp), maxHealth(hp), attackPower(atk), ability(ab) {}

    // ABSTRACTION: pure virtual
    virtual void attack() = 0;

    void normalAttack(Dragon* enemy) {
        cout << "[Polymorphism] Normal attack\n";
        enemy->takeDamage(attackPower);
    }

    void specialAttack(Dragon* enemy) {
        if (ability.canUse()) {
            ability.use();
            enemy->takeDamage(ability.getDamage());
        } else { cout << "Cooldown!\n"; }
    }

    void takeDamage(int dmg) {
        health -= dmg;
        if (health < 0) health = 0;
    }

    bool   isAlive()  { return health > 0; }
    string getName()  { return name; }
    void   reduceCooldown() { ability.reduceCooldown(); }
    virtual ~Dragon() {}
};
// ============================================
// INHERITANCE + POLYMORPHISM
// All three extend Dragon (IS-A)
// Each OVERRIDES attack() differently
// ============================================

class FireDragon : public Dragon {
public:
    FireDragon(string n)
        : Dragon(n, 300, 80, Ability("Inferno Blast", 120, 2)) {}
    void attack() override {   // POLYMORPHISM
        cout << "[FireDragon] Fire attack!\n";
    }
};

class IceDragon : public Dragon {
public:
    IceDragon(string n)
        : Dragon(n, 350, 70, Ability("Frozen Spike", 100, 2)) {}
    void attack() override {
        cout << "[IceDragon] Ice attack!\n";
    }
};

class ForestDragon : public Dragon {
public:
    ForestDragon(string n)
        : Dragon(n, 400, 60, Ability("Poison Vines", 90, 1)) {}
    void attack() override {
        cout << "[ForestDragon] Nature attack!\n";
    }
};

// ENCAPSULATION + ASSOCIATION
class Rider {
private: // ENCAPSULATION
    string name;
    Dragon* dragon; // ASSOCIATION
    int level, gold;
    bool exitedToMenu;
public:
    Rider(string n, Dragon* d)
        : name(n), dragon(d), level(1), gold(0), exitedToMenu(false) {
        cout << "[Encapsulation] Rider " << name << " created\n";
    }
    void battle(Dragon* enemy);
    bool didExit() { return exitedToMenu; }
};
int main() {
    while(true) {
        cout << "====== DRAGON LEGACY ======\n";
        cout << "1. Start Game\n2. Exit\n";
        int menu; cin >> menu;
        if(menu == 2) break;

        string riderName;
        cout << "Enter Rider Name: ";
        cin >> riderName;

        cout << "1.Fire 2.Ice 3.Forest\n";
        int choice; cin >> choice;

        // POLYMORPHISM — base class pointer
        Dragon* player;
        if(choice==1)      player=new FireDragon("Ember");
        else if(choice==2) player=new IceDragon("Frost");
        else               player=new ForestDragon("Verdant");

        Rider rider(riderName, player);

        string quests[] = {"Ashveil","Citadel","Marsh","Lair"};
        for(int i=0; i<4; i++) {
            Dragon* enemy = new FireDragon("Enemy");
            rider.battle(enemy);
            delete enemy;
            if(rider.didExit() || !player->isAlive()) break;
        }
        delete player; // MEMORY MANAGEMENT
    }
    return 0;
}
